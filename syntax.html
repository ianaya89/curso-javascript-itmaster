First Class Citizen

            
              const add = function(x, y) {
                if (x && y ) {
                  return x + y;
                }

                return 0;
              };

              const increment = function (x) {
                return x += 1;
              };

              function addAndIncrement(x, y, addFn, incrementFn) {
                const a = addFn(x, y);
                const b = incrementFn(a);

                return b;
              }

              addAndIncrement(1, 2, add, increment); // 4

Closure

            
              function wrapper(value) {
                const localVar = `[${value}]`;

                return function() {
                  return localVar;
                }
              }

              const wrap1 = wrapper(1);
              const wrap2 = wrapper(2);

              wrap1(); // [1]
              wrap2(); // [2]

Herencia

            
              // padre
              const Person = function(name, id) {
                this.name = name;
                this.id = id;
              }

              Person.prototype.sayHello = function() {
                return 'Hello my name is: ' + this.name;
              }

              // hijo
              const Child = function() {
                Person.call(this); //super
              }

              // herencia
              Child.prototype = Object.create(Person.prototype);
              Child.prototype.constructor = Child;

              Child.prototype.eay = function() {
                return 'eating...';
              }

              const jack = new Child('Jack', 1234);

              console.log(jack.sayHello());
              console.log(jack.eat());
